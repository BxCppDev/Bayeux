154  /* Utility to enable auto-(un)registration of the Type class in the
155  * the system factory register:
156  */
159  template <class BaseType, class DerivedType>
160  class _system_factory_registrator {
161  public:
162  // Constructor
163  _system_factory_registrator(const std::string& type_id) {
164  type_id_ = type_id;
165  this->_trigger_factory_registration_();
166  }
167  // Return registered type id
168  const std::string& get_type_id()const
169  {
170  return type_id_;
171  }
172  // Destructor
173  ~_system_factory_registrator() {
174  this->_trigger_factory_unregistration_();
175  }
176 
177  private:
178  // Factory registration
179  void _trigger_factory_registration_() {
180  bool fatal = !boost::is_base_of<BaseType, DerivedType>::value;
181  DT_THROW_IF (fatal,
182  std::logic_error,
183  "Class ID '" << type_id_ << "' cannot be registered in register '"
184  << BaseType::grab_system_factory_register().get_label() << "' !");
185  BaseType::grab_system_factory_register().registration(type_id_,
186  boost::factory<DerivedType*>());
187  }
188 
189  // Factory unregistration
190  void _trigger_factory_unregistration_() {
191  if (BaseType::grab_system_factory_register().has(type_id_)) {
192  BaseType::grab_system_factory_register().unregistration(type_id_);
193  }
194  }
195 
196  private:
197  std::string type_id_;
198  };
