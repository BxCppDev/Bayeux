171  template <typename T>
172  const typename handle_pool<T>::handle_type& handle_pool<T>::create() {
173 #ifdef DATATOOLS_HANDLE_POOL_DEVEL
174  DT_LOG_TRACE_ENTERING(datatools::logger::PRIO_ALWAYS);
175  DT_LOG_TRACE(datatools::logger::PRIO_ALWAYS, "buffer size is " << buffer.size());
176  DT_LOG_TRACE(datatools::logger::PRIO_ALWAYS, "number of used element " << _number_of_used_item);
177 #endif
178 
179  if (number_of_used_item_ < buffer_.size()) {
180 #ifdef DATATOOLS_HANDLE_POOL_DEVEL
181  DT_LOG_TRACE(datatools::logger::PRIO_ALWAYS, "Use a pre-allocated element @ position "
182  << number_of_used_item_);
183  DT_LOG_TRACE_EXITING(datatools::logger::PRIO_ALWAYS);
184 #endif
185  return buffer_[number_of_used_item_++];
186  }
187 
188  element_type *ptr = new element_type;
189  buffer_.push_back(handle_type(ptr));
190  number_of_used_item_++;
191 #ifdef DATATOOLS_HANDLE_POOL_DEVEL
192  DT_LOG_TRACE(datatools::logger::PRIO_ALWAYS, "Use a newly allocated element @ position "
193  << (number_of_used_item_ - 1));
194  DT_LOG_TRACE_EXITING(datatools::logger::PRIO_ALWAYS);
195 #endif
196  return buffer_.back();
197  }

200  template <typename T>
201  void handle_pool<T>::dump(std::ostream& out,
202  const std::string& title,
203  const std::string& indent,
204  bool abridged) const {
205  if (!title.empty()) {
206  out << indent << title << " : " << std::endl;
207  }
208  out << indent << "|-- " << "Capacity : " << this->get_capacity() << std::endl;
209  out << indent << "|-- " << "# Used items : " << number_of_used_item_ << std::endl;
210  out << indent << "`-- " << "Items @ " << std::endl;
211  for (size_t i = 0; i < buffer_.size(); ++i) {
212  if (abridged) {
213  if (i >= 10) continue;
214  }
215  out << indent << " ";
216  const handle_type& h = buffer_[i];
217  if (i < buffer_.size() - 1) {
218  out << "|-- ";
219  } else {
220  out << "`-- ";
221  }
222  out << "index " << i << " : ";
223  if (!h) {
224  out << "No element!" << std::endl;
225  }
226  else {
227  out << "@ " << &h.get() << " [unique=" << h.unique() << "]"<< std::endl;
228  }

233  template <typename T>
234  void handle_pool<T>::resize_impl(size_t size) {
235 #ifdef DATATOOLS_HANDLE_POOL_DEVEL
236  DT_LOG_TRACE_ENTERING(datatools::logger::PRIO_ALWAYS);
237 #endif
238 
239  size_t current_size = buffer_.size();
240 
241 #ifdef DATATOOLS_HANDLE_POOL_DEVEL
242  DT_LOG_TRACE(datatools::logger::PRIO_ALWAYS, "current_size = " << current_size);
243 #endif
244 
245  if (size <= current_size) {
246 #ifdef DATATOOLS_HANDLE_POOL_DEVEL
247  DT_LOG_TRACE(datatools::logger::PRIO_ALWAYS, "Capacity is enough for " << size);
248 #endif
249  return;
250  }
251 
252  buffer_.reserve(size);
253  for (size_t i = current_size; i < size; ++i) {
254 #ifdef DATATOOLS_HANDLE_POOL_DEVEL
255  DT_LOG_TRACE(datatools::logger::PRIO_ALWAYS, "Add a handle with a new element...");
256 #endif
257  buffer_.push_back(handle_type(new element_type));
258  }
259 
260  number_of_used_item_ = 0;
261 
262 #ifdef DATATOOLS_HANDLE_POOL_DEVEL
263  DT_LOG_TRACE_EXITING(datatools::logger::PRIO_ALWAYS);
264 #endif
265 
266  return;
267  }
