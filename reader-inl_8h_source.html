54  template<typename T>
55  int reader::load(T& data_, const std::string& label_, int64_t nentry_) {
56  DT_THROW_IF(!this->is_opened(),
57  std::logic_error,
58  "Operation prohibited; file is not opened !");
59 
60  store_info *ptr_si = this->_get_store_info(label_);
61  if (!ptr_si) {
62  DT_THROW_IF(!label_.empty (),
63  std::logic_error,
64  "No source store with label '" << label_ << "' !");
65  // if we do not allow automatic store, this is a critical error:
66  DT_THROW_IF(!_allow_automatic_store_,
67  std::logic_error,
68  "No source store is selected nor default store is available !");
69  ptr_si = _automatic_store_;
70  }
71  return this->_at_load<T>(data_, ptr_si, nentry_);
72  }

74  template<class T>
75  int reader::_at_load(T& data_, store_info *ptr_si_, int64_t nentry_) {
76  DT_LOG_TRACE(this->get_logging_priority(),"Entering...");
77  store_info& si = *ptr_si_;
78 
79  if (_check_serial_tag_) {
80  // We check if the serialization tag from the store matches the
81  // data's one:
82  if (si.has_dedicated_serialization_tag()) {
83  // 2013-02-19 FM : change the way we check the serial tag :
84  //if (data_.get_serial_tag () != si.get_serialization_tag ())
85  DT_THROW_IF(!datatools::check_serial_tag<T>(si.get_serialization_tag ()),
86  std::logic_error,
87  "Data serialization tag '" << data_.get_serial_tag()
88  << "' does not match source store's serialization tag '" << si.get_serialization_tag () << "' !");
89  }
90  }
91 
92  DT_THROW_IF(si.number_of_entries == 0,
93  std::logic_error,
94  "Source store '" << si.label << "' has no entry !");
95  int64_t nentry = nentry_;
96  if (nentry >= 0) {
97  // check overflow:
98  DT_THROW_IF(nentry_ >= si.number_of_entries,
99  std::logic_error,
100  "Source store '" << si.label << "' has "
101  << "no serialized entry at index '" << nentry_ << "' !");
102  } else {
103  // if nentry_ < 0: use entry index relative to the current entry
104  // position
105  if (si.current_entry < 0) {// at rewind position
106  // start with first entry:
107  nentry = 0;
108  } else {
109  // try next entry:
110  nentry = si.current_entry + 1;
111  }
112  }
113 
114  // read this tree entry in the ROOT I/O system:
115  si.record.reset();
116  int ret = si.tree->GetEntry(nentry, 1); // -> 1 for all branches
117  if (!ret) {
118  DT_THROW_IF(true,
119  std::logic_error,
120  "No entry '"
121  << nentry << "' at entry # " << nentry
122  << " in source store labelled '" << si.label.c_str()
123  << "' from file '" << _filename << "' !");
124  } else if (ret == -1) {
125  DT_THROW_IF(true,
126  std::logic_error,
127  "An I/O error occurs from entry '"
128  << nentry
129  << "' in source store labelled '" << si.label.c_str()
130  << "' from file '" << _filename << "' !");
131  } else {
132  si.current_entry = nentry;
133  }
134 
135  if (_check_serial_tag_) {
136  /* We may be confused with stores without dedicated serialization tag.
137  * Here we test if data and the entry's serial tags match:
138  */
139  if (!si.has_dedicated_serialization_tag()) {
140  // check serial tag associated to the buffered binary archive:
141  std::string serial_tag = si.record.fSerialTag.Data();
142  // 2013-02-19 FM : change the way we check
143  // if (data_.get_serial_tag () != serial_tag)
144  DT_THROW_IF(!datatools::check_serial_tag<T>(serial_tag),
145  std::logic_error,
146  "Entry '" << nentry
147  << "' with serial tag '" << serial_tag
148  << "' in (mixed) source store labelled '" << si.label.c_str()
149 
150  << "' from file '" << _filename
151  << "' does not match the requested '"
152  << data_.get_serial_tag() << "' data type !");
153  }
154  }
155 
156  // Deserialize from the archive:
157  boost::iostreams::stream<boost::iostreams::array_source>
158  input_stream(si.record.fDataBuffer.fArray,si.record.fDataBuffer.fN);
159  // 2011-06-16 FM: restored
160  if (this->is_format_pba()) {
161  //boost::archive::portable_binary_iarchive ia (input_stream);
162  eos::portable_iarchive ia(input_stream);
163  ia >> data_;
164  }
165  if (this->is_format_text()) {
166  input_stream.imbue (*_locale);
167  boost::archive::text_iarchive ia (input_stream);
168  ia >> data_;
169  }
170 
171  _current_store = &si;
172  DT_LOG_TRACE(this->get_logging_priority(),"Exiting.");
173  return 0;
174  }
