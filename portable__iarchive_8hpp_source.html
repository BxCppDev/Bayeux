344  template <typename T>
345  typename boost::enable_if<boost::is_integral<T> >::type
346  load(T & t, dummy<2> = 0)
347  {
348  // get the number of bytes in the stream
349  if (signed char size = load_signed_char())
350  {
351  // check for negative value in unsigned type
352  if (size < 0 && boost::is_unsigned<T>::value)
353  throw portable_archive_exception();
354 
355  // check that our type T is large enough
356  else if ((unsigned) abs(size) > sizeof(T))
357  throw portable_archive_exception(size);
358 
359  // reconstruct the value
360  T temp = size < 0 ? -1 : 0;
361  load_binary(&temp, abs(size));
362 
363  // load the value from little endian - is is then converted
364  // to the target type T and fits it because size <= sizeof(T)
365  t = endian::load_little_endian<T, sizeof(T)>(&temp);
366  }
367 
368  else t = 0; // zero optimization
369  }

398  template <typename T>
399  typename boost::enable_if<boost::is_floating_point<T> >::type
400  load(T & t, dummy<3> = 0)
401  {
402  typedef typename fp::detail::fp_traits<T>::type traits;
403 
404  // if you end here there are three possibilities:
405  // 1. you're serializing a long double which is not portable
406  // 2. you're serializing a double but have no 64 bit integer
407  // 3. your machine is using an unknown floating point format
408  // after reading the note above you still might decide to
409  // deactivate this static assert and try if it works out.
410  typename traits::bits bits;
411 #ifdef __clang__
412 #pragma clang diagnostic push
413 #pragma clang diagnostic ignored "-Wunused-local-typedef"
414 #endif
415  BOOST_STATIC_ASSERT(sizeof(bits) == sizeof(T));
416  BOOST_STATIC_ASSERT(std::numeric_limits<T>::is_iec559);
417 #ifdef __clang__
418 #pragma clang diagnostic pop
419 #endif
420 
421  load(bits);
422  traits::set_bits(t, bits);
423 
424  // if the no_infnan flag is set we must throw here
425  if (get_flags() & no_infnan && !fp::isfinite(t))
426  throw portable_archive_exception(t);
427 
428  // if you end here your floating point type does not support
429  // denormalized numbers. this might be the case even though
430  // your type conforms to IEC 559 (and thus to IEEE 754)
431  if (std::numeric_limits<T>::has_denorm == std::denorm_absent
432  && fp::fpclassify(t) == (int) FP_SUBNORMAL) // GCC4
433  throw portable_archive_exception(t);
434  }
