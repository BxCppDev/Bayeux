550  template <class ran_func>
551  void randomize_direction(ran_func & ran_, vector_3d & ran_dir_)
552  {
553  double phi = 2. * M_PI * ran_();
554  double cos_theta = -1 + 2 * ran_();
555  double sin_theta = std::sqrt(1. - cos_theta * cos_theta);
556  double x = sin_theta * std::cos(phi);
557  double y = sin_theta * std::sin(phi);
558  double z = cos_theta;
559  ran_dir_.set(x, y, z);
560  }
561 
571  template <class ran_func>
572  vector_3d randomize_direction(ran_func & ran_)
573  {
574  vector_3d dir;
575  randomize_direction(ran_, dir);
576  return dir;
577  }

589  template <class ran_func>
590  void randomize_orthogonal_direction(ran_func & ran_,
591  const vector_3d & dir_,
592  vector_3d & ran_dir_)
593  {
594  double theta = 2. * M_PI * ran_();
595  double dx = cos(theta);
596  double dy = sin(theta);
597  double dz = 0.0;
598  double dir_theta = dir_.theta();
599  double dir_phi = dir_.phi();
600  rotation_3d dir_rot;
601  create_rotation_3d(dir_rot, dir_phi, dir_theta, 0.0);
602  rotation_3d dir_inverse_rot;
603  dir_inverse_rot = dir_rot.inverse();
604  vector_3d v(dx, dy, dz);
605  ran_dir_ = v.transform(dir_inverse_rot);
606  return;
607  }
