189  out << indent << i_tree_dumpable::inherit_tag(inherit)
190  << "Registered factories : " << registered_.size() << std::endl;
191 
192  for (typename factory_map_type::const_iterator i = registered_.begin();
193  i != registered_.end ();
194  ++i) {
195  out << a_indent;
196  out << i_tree_dumpable::inherit_skip_tag(inherit);
197  typename factory_map_type::const_iterator j = i;
198  j++;
199  if (j == registered_.end()) {
200  out << i_tree_dumpable::last_tag;
201  } else {
202  out << i_tree_dumpable::tag;
203  }
204 
205  out << "ID: \"" << i->first << "\" @ " << &i->second << std::endl;
206  }
