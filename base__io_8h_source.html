165  template <class T>
166  bool base_io::has_store_with_matching_serial_tag(const std::string& label_) const {
167  store_info_dict_type::const_iterator found = _store_infos.find(label_);
168  if (found == _store_infos.end()) return false;
169  const store_info& the_si = found->second;
170  if (the_si.has_dedicated_serialization_tag()) {
171  return datatools::check_serial_tag<T>(the_si.get_serialization_tag());
172  }
173  return false;
174  }
