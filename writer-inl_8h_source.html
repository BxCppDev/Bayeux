47  template <typename T>
48  int writer::store(const T& a_data, const std::string& a_label) {
49  DT_THROW_IF(!this->is_opened(),
50  std::logic_error,
51  "Operation prohibited; file is not opened !");
52  store_info *ptr_si = this->_get_store_info(a_label);
53  if (!ptr_si) {
54  if (a_label.empty()) {
55  // if we do not allow automatic store, this is a critical error:
56  DT_THROW_IF(!_allow_automatic_store_,
57  std::logic_error,
58  "No target store is selected nor target !");
59  ptr_si = this->_add_store(store_info::constants::automatic_store_label(),
60  a_data.get_serial_tag(),
61  store_info::constants::default_store_buffer_size());
62  } else {
63  ptr_si = this->_add_store(a_label, a_data.get_serial_tag(),
64  store_info::constants::default_store_buffer_size());
65  }
66  }
67  // Final check:
68  DT_THROW_IF(ptr_si == 0,
69  std::logic_error,
70  "Could not determine any store to save data !");
71  DT_LOG_DEBUG(this->get_logging_priority(),
72  "ptr_si = " << std::hex << ptr_si << std::dec);
73  DT_LOG_DEBUG(this->get_logging_priority(),
74  "Using store with label '" << ptr_si->label << "'...");
75  if (this->get_logging_priority() >= datatools::logger::PRIO_DEBUG) {
76  ptr_si->tree->Print ();
77  }
78  return this->_at_store<T>(a_data, ptr_si);
79  }

81  template <typename T>
82  int writer::_at_store(const T& a_data, store_info *a_store_info) {
83  DT_LOG_TRACE(this->get_logging_priority(),"Entering...");
84  store_info *ptr_si = a_store_info;
85  // First serialized object sets the serialization tag for this store:
86  if (ptr_si->serialization_tag ==
87  store_info::constants::postponed_dedicated_serial_tag_label()) {
88  ptr_si->serialization_tag = a_data.get_serial_tag();
89  }
90  // Else if the store has a dedicated serialization tag:
91  else if (ptr_si->has_dedicated_serialization_tag()) {
92  // Check the data serialization tag matches that requested by the store:
93  DT_THROW_IF(a_data.get_serial_tag() != ptr_si->get_serialization_tag(),
94  std::logic_error,
95  "Serialization tag mismatch ! "
96  << "Attempt to store an object with `"
97  << a_data.get_serial_tag ()
98  << "' serialization tag "
99  << "in the store labelled '"
100  << ptr_si->label
101  << "' with dedicated `"
102  << ptr_si->get_serialization_tag ()
103  << "' serialization tag !");
104  }
105 
106  // Prepare the (std::vector<char>) buffer to host the binary archive
107  // as a sequence of bytes:
108  typedef std::vector<char> buffer_type;
109 
110  /* Clear the buffer of characters for streaming but
111  * intrinsic capacity is kept as is (this is a behaviour of vector
112  * class from the STL library):
113  */
114  if (!ptr_si->buffer.empty()) {
115  ptr_si->buffer.clear();
116  } else if (ptr_si->buffer.capacity() == 0) {
117  // Ensure minimum starting capacity of the buffer of characters for
118  // streaming in order to optimize possible memory reallocation:
119  ptr_si->buffer.reserve(store_info::constants::default_stream_buffer_size());
120  }
121 
122  // Archiving is redirected to the buffer:
123  namespace io = boost::iostreams;
124  io::stream<io::back_insert_device<buffer_type> > output_stream(ptr_si->buffer);
125  // 2011-06-16 FM: restored
126  if (this->is_format_pba()) {
127  //boost::archive::portable_binary_oarchive oa (output_stream);
128  eos::portable_oarchive oa(output_stream);
129  oa << a_data;
130  }
131  if (is_format_text()) {
132  output_stream.imbue(*_locale);
133  boost::archive::text_oarchive oa(output_stream);
134  oa << a_data;
135  }
136  output_stream.flush();
137 
138  // Now the buffer contains the final sequence of bytes corresponding to
139  // the serialized output binary archive:
140  DT_LOG_DEBUG(this->get_logging_priority(),
141  "Buffer size = "
142  << ptr_si->buffer.size()
143  << " buffer capacity = "
144  << ptr_si->buffer.capacity());
145  // Prepare the container interface to be streamed using the ROOT I/O system:
146  ptr_si->record.fSerialTag = a_data.get_serial_tag().c_str();
147  ptr_si->record.fVersionTag = 0; // not used
148  ptr_si->record.fDataBuffer.fN = ptr_si->buffer.size();
149  ptr_si->record.fDataBuffer.fArray = &(ptr_si->buffer[0]);
150  // Store this tree entry in the ROOT I/O system:
151  ptr_si->tree->Fill ();
152  // Reset the ROOT data buffer:
153  ptr_si->record.fDataBuffer.fN = 0;
154  ptr_si->record.fDataBuffer.fArray = 0;
155  // Update some counters:
156  ptr_si->number_of_entries = ptr_si->tree->GetEntries();
157  ptr_si->current_entry = ptr_si->number_of_entries - 1;
158  // Update the 'current store' pointer:
159  _current_store = ptr_si;
160  DT_LOG_TRACE(this->get_logging_priority(),"Exiting.");
161  return 0;
162  }
