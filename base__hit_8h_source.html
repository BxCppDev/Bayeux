242 
244  bool operator () (const base_hit & a_hit) const
245  {
246  if (! a_hit.get_auxiliaries ().has_key (key_)) return false;
247  if (! a_hit.get_auxiliaries ().is_string (key_)) return false;
248  const std::string & str = a_hit.get_auxiliaries ().fetch_string (key_);
249  return (std::find (values_.begin (),
250  values_.end (),
251  str) != values_.end ());
252  }
