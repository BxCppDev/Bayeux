254  void store(const Data& data) {
255  DT_THROW_IF (!this->is_write(),
256  std::logic_error,

267  template <typename Data>
268  void load(Data& data) {
269  DT_THROW_IF (!this->is_read(),
270  std::logic_error,
271  "Not a reader factory!");
272  DT_THROW_IF (in_fs_ == 0,
273  std::runtime_error,
274  "No input stream!");
275 
276  // 2008-10-03 FM: add EOF check code
277  if (*in_fs_) {
278  // 2009-03-08 FM: Is it possible to have better EOF check code
279  // here for unzipped text and XML archives?
280  char c = 0;
281  in_fs_->get(c);
282  if (in_fs_) {
283  in_fs_->putback(c);
284  }
285  }
286 
287  if (! *in_fs_) {
288  DT_THROW_IF (in_fs_->eof(),
289  std::runtime_error,
290  "Input stream at EOF!");
291  DT_THROW_IF (in_fs_->fail(),
292  std::runtime_error,
293  "Input stream in fail status!");
294  DT_THROW_IF (in_fs_->bad(),
295  std::runtime_error,
296  "Input stream in bad status!");
297  }
298 
299  try {
300  if (itar_ptr_ != 0) {
301  this->load_text<Data>(*itar_ptr_, data);
302  *in_fs_ >> std::ws;
303  } else if (ixar_ptr_ != 0) {
304  this->load_xml<Data>(*ixar_ptr_, data);
305  *in_fs_ >> std::ws;
306  } else if (ibar_ptr_ != 0) {
307  this->load_binary<Data>(*ibar_ptr_, data);
308  }
309  }
310  // 2011-02-25 FM:
311  catch (boost::archive::archive_exception& x) {
312  throw;
313  }
314  catch (std::exception& x) {
315  throw;
316  }
317  catch (...) {
318  DT_LOG_WARNING(get_logging_priority(),
319  "Cannot load data from archive: "
320  << "unexpected exception" << "!");
321  DT_THROW(std::runtime_error, "Internal exception!");
322  }
323 
324  if (! *in_fs_) {
325  if (in_fs_->fail()) {
326  DT_LOG_WARNING(get_logging_priority(),
327  "Input stream is now in fail status!");
328  }
329  if (in_fs_->eof()) {
330  DT_LOG_WARNING(get_logging_priority(),
331  "Input stream is now in EOF status!");
332  }
333  if (in_fs_->bad()) {
334  DT_LOG_WARNING(get_logging_priority(),
335  "Input stream is now in bad status!");
336  }
337  }
338  }
