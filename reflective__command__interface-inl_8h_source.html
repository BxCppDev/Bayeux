105  template <class Type>
106  void reflective_command_interface<Type>::_add_commands()
107  {
108  const camp::Class * metaclass = camp::classByTypeSafe<Type>();
109  DT_THROW_IF(metaclass == nullptr, std::logic_error,
110  "Class '" << typeid(Type).name() << "' has no reflection support!");
111 
112  for (std::size_t ifunc = 0; ifunc < metaclass->functionCount(); ifunc++) {
113  const camp::Function & func = metaclass->function(ifunc);
114  if (!func.hasTag(datatools::introspection::function_builder_flag())) {
115  continue;
116  }
117  if (_disabled_commands_.count(func.name()) != 0) {
118  continue;
119  }
120  datatools::ui::reflective_command<Type> * cmd
121  = new datatools::ui::reflective_command<Type>(this->reflective_command_interface<Type>::_grab_target(), func.name());
122  cmd->initialize_simple();
123  datatools::ui::reflective_command_interface<Type>::add_command(cmd, func.name());
124  }
125 
126  return;
127  }
