

Example :

Explicit preparation of Cadfael dependencies :

{{{
bash$ source <Cadfael install path>/etc/cadfael_setup.sh
bash$ do_cadfael_all_setup
}}}

Configure :

{{{
bash$ ./pkgtools.d/pkgtool configure --workspace-dir /scratch/sw/Bayeux
}}}

or for the legacy version :
{{{
bash$ ./pkgtools.d/pkgtool configure \
  --workspace-dir /scratch/sw/Bayeux-legacy \
  --config legacy --bayeux-label legacy
}}}

or for the trunk version (not working):
{{{
bash$ ./pkgtools.d/pkgtool configure \
  --workspace-dir /scratch/sw/Bayeux-trunk \
  --config trunk
}}}

With explicit path to Cadfael FindModule:
{{{
bash$ ./pkgtools.d/pkgtool configure \
  --workspace-dir /scratch/sw/Bayeux \
  --cadfael-dir /scratch/sw/Cadfael/install/Cadfael-0.1.0/share/cmake/Modules
}}}

{{{
bash$ ./pkgtools.d/pkgtool configure \
        --workspace-dir /scratch/sw/Bayeux \
        --config [last_tags|trunk|default]
}}}

{{{
bash$ ./pkgtools.d/pkgtool configure \
        --config "0.9.0" \
        --workspace-dir /scratch/sw/Bayeux \
        --with-all \ 
        --without-genbb_help 
}}}

{{{
bash$ ./pkgtools.d/pkgtool --devel  \
        configure \
        --config user --user-config-file "examples/bayeux_config_1.lis" \
        --workspace-dir /scratch/sw/Bayeux  
}}}

{{{
bash$ ./pkgtools.d/pkgtool configure \
        --install-prefix     /scratch/sw/Bayeux/Bayeux-0.9.0/install \
        --ep-build-directory /scratch/sw/Bayeux/Bayeux-0.9.0/build 
}}}


Build :

{{{
bash$ ./pkgtools.d/pkgtool build 
}}}


Install :

{{{
bash$ ./pkgtools.d/pkgtool install
}}}


Setup :

{{{
bash$ export BAYEUX_PRE_SETUP_SCRIPT="${CADFAEL_ETC_DIR}/cadfael_setup.sh"
bash$ export BAYEUX_PRE_SETUP_FUNCTION="do_cadfael_all_setup"
bash$ source /scratch/sw/Bayeux/Bayeux-0.9.0/Install-Linux-i686/etc/bayeux_setup.sh 
bash$ do_bayeux_all_setup
}}}


in '~/.bashrc' :

>>>
export BAYEUX_PRO_INSTALL_DIR=/scratch/sw/Bayeux/Bayeux-0.9.0/Install-Linux-i686
function do_bayeux_pro_setup ()
{
  source ${BAYEUX_PRO_INSTALL_DIR}
  do_bayeux_all_setup
  return 0
}

export BAYEUX_TRUNK_INSTALL_DIR=/scratch/sw/Bayeux-trunk/Bayeux-0.9.1/Install-Linux-i686
function do_bayeux_trunk_setup ()
{
  source ${BAYEUX_TRUNK_INSTALL_DIR}
  do_bayeux_all_setup
  return 0
}
<<<
