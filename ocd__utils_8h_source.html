165  template<class T>
166  void ocd_registration::registration(const std::string & class_id_,
167  const ocd_handle_type & handle_) {
168  DT_LOG_TRACE(_logging_, "Attempt to register class ID '" << class_id_ << "' !");
169  if (has_id(class_id_)) {
170  std::ostringstream message;
171  message << "datatools::detail::ocd::ocd_registration::registration: "
172  << "Class ID '" << class_id_ << "' is already registered !";
173  throw std::logic_error(message.str());
174  }
175  {
176  entry_type dummy;
177  _dict_[class_id_] = dummy;
178  }
179  entry_type & e = _dict_[class_id_];
180  e.handle = handle_;
181  DT_LOG_TRACE(_logging_, "Registration of class ID '" << class_id_ << "' was successful!");
182  return;
183  }

185  template<class T>
186  void ocd_registration::unregistration(const std::string & class_id_) {
187  DT_LOG_TRACE(_logging_, "Attempt to unregister class ID '" << class_id_ << "' !");
188  ocd_dict_type::iterator found = _dict_.find(class_id_);
189  if (found == _dict_.end()) {
190  DT_LOG_ERROR(_logging_, "Class ID '" << class_id_ << "' is not registered !");
191  return;
192  // throw std::logic_error(message.str());
193  }
194  if (! found->second.handle.unique()) {
195  DT_LOG_FATAL(_logging_,
196  "OCD registration entry for class ID '" << class_id_ << "' is not uniquely referenced !"
197  << " TO BE FIXED!");
198  }
199  _dict_.erase(found);
200  DT_LOG_TRACE(_logging_, "Unregistration of class ID '" << class_id_ << "' was successful!");
201  return;
202  }

212  template <class T>
213  class system_factory_registrar : public base_system_factory_registrar
214  {
215  public:
216  system_factory_registrar(const std::string & class_id_) {
217  _logging_ = ::datatools::logger::PRIO_FATAL;
218  char * docd_logging = getenv("DATATOOLS_OCD_DEVEL_LOGGING");
219  if (docd_logging) {
220  std::string ocd_logging_label = docd_logging;
221  _logging_ = ::datatools::logger::get_priority(ocd_logging_label);
222  if (_logging_ == ::datatools::logger::PRIO_UNDEFINED) {
223  _logging_ = ::datatools::logger::PRIO_FATAL;
224  }
225  }
226  DT_LOG_TRACE(_logging_, "Creation of OCD system_factory_registrar for class '" << class_id_ << "'...");
227  DT_THROW_IF(class_id_.empty(), std::logic_error, "Class ID is empty!");
228  ocd_registration & ocd_reg = ocd_registration::grab_system_registration();
229  ocd_registration::ocd_handle_type h;
230  h.reset(new object_configuration_description);
231  object_configuration_description & the_ocd = *h.get();
232  bool support = load_ocd<T>(the_ocd);
233  if (support) {
234  ocd_reg.registration<T>(class_id_, h);
235  _class_id_ = class_id_;
236  }
237  }
238  virtual ~system_factory_registrar() {
239  DT_LOG_TRACE(_logging_, "Destruction of OCD system_factory_registrar for registered class '" << _class_id_ << "'...");
240  ocd_registration & ocd_reg = ocd_registration::grab_system_registration();
241  if (! _class_id_.empty() && ocd_reg.has_id(_class_id_)) {
242  ocd_reg.unregistration<T>(_class_id_);
243  }
244  }
245  private:
246 
247  logger::priority _logging_;
248  std::string _class_id_;
249  };
