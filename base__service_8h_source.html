156  template<class T>
157  T& grab(service_dict_type & services_, const std::string& service_name_)
158  {
159  base_service & srvc = grab_service(services_, service_name_);
160  const std::type_info& ti = typeid(T);
161  const std::type_info& tf = typeid(srvc);
162  DT_THROW_IF(ti != tf, std::logic_error, "Service '" << service_name_ << "' is not of requested type!");
163  return dynamic_cast<T&>(srvc);
164  }
